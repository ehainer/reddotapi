def assign!(name_or_responsibility)
  new_responsibility = Responsibility.fetch!(name_or_responsibility)

  transaction do
    self.roles = []
    self.responsibility = new_responsibility
    new_responsibility.roles.each { |r| add_role(r) }
  end
end

def available_tasks
  facilities = facility_group.empty? ? Facility.all : facility_group

  facility_ids = facilities.includes(:units).map(&:id)
  unit_ids = Facility.unit_ids_from_facilities(facilities)
  Task.created.find_for_any(facility_ids: facility_ids, unit_ids: unit_ids)
end

def facilities
  results = accounts.joins(occupancies: { unit: [:facility] })
  results = results.where('accounts.id' => account)
  results = results.group('facilities.id, occupancies.id').order('occupancies.id ASC').pluck('facilities.id').uniq
  Facility.find(results)
end

def units(*facilities)
  # Form array of facility ids to filter by
  facilities = facilities.flatten.map { |facility| facility.is_a?(Facility) ? facility.id : facility }

  results = accounts.joins(occupancies: { unit: [:facility] })
  results = results.where('facilities.id' => facilities) unless facilities.empty?
  results = results.pluck('units.id').uniq
  Unit.find(results)
end
