def self.setup(account)
  account.facilities(false).each do |facility|
    unless Gate.exists?(account: account, facility: facility)
      Gate.create(account: account, facility: facility)
    end
  end
end

def generate(length=6, **options)
  candidate = { facility_id: facility_id, account_id: account_id, code: candidate(length, options.with_indifferent_access) }

  while Gate.exists?(candidate)
    candidate = { facility_id: facility_id, account_id: account_id, code: candidate(length, options.with_indifferent_access) }
  end

  candidate[:code]
end
