def self.fetch!(name_or_responsibility)
  if name_or_responsibility.is_a?(Symbol) || name_or_responsibility.is_a?(String)
    find_by!(name: name_or_responsibility)
  elsif name_or_responsibility.is_a?(Responsibility)
    name_or_responsibility
  else
    raise InvalidResponsibilityError, "#{name_or_responsibility} is not a valid responsibility"
  end
end
