def activate(now=false)
  activate!(now)
rescue InvalidActivationWindow
  errors.add(:base, '^Occupancy cannot be activated at this time'._)
  false
rescue AlreadyActivated
  errors.add(:base, '^Occupancy has already been activated'._)
  false
end

def activate!(now=false)
  raise InvalidActivationWindow if !Time.current.between?(moved_in_at.utc.beginning_of_day, moved_in_at.utc.end_of_day) && !now
  raise AlreadyActivated if active?
  update!(active: true, signed_up_at: Time.current)
end

def occupied?(when_at=moved_in_at)
  return true unless when_at.is_a?(Time) # If not an instance of {Time}
  !Occupancy.where(unit: unit).where('moved_out_at > ? OR moved_out_at = ?', when_at.utc, nil).count.zero?
end
