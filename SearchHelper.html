def can_show_per_page?(page)
  if @search.present?
    index = SearchService::PAGING_OPTIONS.index(page.to_i) || 0
    current_count = params[:search][:per_page] || SearchService::PAGING_OPTIONS.first
    page.to_i < current_count.to_i || @search.documents.hits.total_entries > (index.zero? ? SearchService::PAGING_OPTIONS.first : SearchService::PAGING_OPTIONS[index - 1])
  else
    false
  end
end
