def initialize(user)
  user ||= User.new
  @user = user

  if @user.has_role?(:admin)
    admin_permissions
  end

  if @user.employee?
    employee_permissions
  end

  if @user.has_role?(:csr)
    csr_permissions
  end

  if @user.has_role?(:fs)
    fs_permissions
  end

  if @user.has_role?(:fs) && @user.has_role?(:supervisor)
    fom_permissions
  end
end

def employee_permissions
  can :manage, User do |u|
    u.customer? || u == @user
  end

  can :manage, Reminder do |r|
    r.user == @user
  end

  can :view, :employee_dashboards
  can :show, Task
end

def fom_permissions
  can :manage, User do |u|
    u.has_role?(:fs) && !u.has_role?(:supervisor)
  end

def fs_permissions
  can :show, Facility do |f|

  can :manage, Gate

  can :read, :facility_status
end
