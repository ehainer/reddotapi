def find_for_any(**taskable_options)
  # Provide arrays of taskable ID's
  # Example: Task.find_for_any(facility_ids: [1, 2, 3], unit_ids: [4, 5, 6], user_ids: [7, 8, 9])

  first_opt_key = taskable_options.keys.first
  first_opt_value = taskable_options.delete(first_opt_key)

  query = Task.where('taskable_type = ? AND taskable_id IN(?)', key_to_taskable_type(first_opt_key), first_opt_value)

  taskable_options.each do |k, v|
    query = query.or(Task.where('taskable_type = ? AND taskable_id IN(?)', key_to_taskable_type(k), v))
  end

  query
end
