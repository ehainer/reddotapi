def search_address
  if addresses.has_default?
    h.content_tag :div, class: 'search-row-address' do
      address = addresses.default
      "#{address.street}, #{address.city}, #{address.state} #{address.zip}"
    end
  end
end

def search_phone
  if contacts.has_default?
    h.content_tag :div, class: 'search-row-phone' do
      content_tag :a, contacts.default.phone.phony_formatted(normalize: 'US'), href: 'tel:%{number}' % { number: contacts.default.phone.phony_normalized }
    end
  end
end

def search_unit
  h.content_tag :div, class: 'search-row-unit' do
    units = []
    [*1..6].sample.times do
      units << h.link_to(([[*1..99].sample, 'ABCDEFGHIJKLMNOPQRSTUVWXYZ'.split('').sample].join('')), '#')
    end
    units.join(', ').html_safe
  end
end

def unlock_button
  return nil if pending_invitation?

  button_text = pending_unlock? ? 'Resend Unlock Instructions'._ : 'Unlock User'._
  button_to(button_text, lock_path(id), method: :delete, remote: true, form: { data: { dispatch: 'user:ajax:success/unlock_user' } })
end
