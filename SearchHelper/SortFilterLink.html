def sorting_by?(field, dir)
  sorting_by = current_params[:search][:sort_by] || SearchService::DEFAULT_SORT_FIELD
  sorting_dir = current_params[:search][:sort_dir] || SearchService::DEFAULT_SORT_DIRECTION

  field.to_s == sorting_by.to_s && dir.to_s == sorting_dir.to_s
end
