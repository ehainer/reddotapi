def search_address
  if holder.addresses.has_default?
    h.content_tag :div, class: 'search-row-address' do
      address = holder.addresses.default
      "#{address.street}, #{address.city}, #{address.state} #{address.zip}"
    end
  end
end

def search_phone
  if holder.contacts.has_default?
    h.content_tag :div, class: 'search-row-phone' do
      content_tag :a, holder.contacts.default.phone.phony_formatted(normalize: 'US'), href: 'tel:%{number}' % { number: holder.contacts.default.phone.phony_normalized }
    end
  end
end

def search_unit(facility)
  h.content_tag :div, class: 'search-row-unit' do
    facility_units = units(true, facility).map do |unit|
      tooltip = []
      tooltip << "<label>Move In:</label> #{h.local_time(unit.occupancy.moved_in_at, '%b %d, %Y')}"
      tooltip << "<label>Move Out:</label> #{h.local_time(unit.occupancy.moved_out_at, '%b %d, %Y')}" if unit.occupancy.moved_out_at.present?
      h.link_to(unit.unit_number, '#', data: { dispatch: 'tooltip:mouseover/element', tooltip: tooltip.map { |t| "<p>#{t}</p>" }.join })
    end
    facility_units.join(', ').html_safe
  end
end
