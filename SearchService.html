def initialize(params, options={}, &block)
  @params = params.permit(:utf8, search: [:q, :sort_dir, :sort_by, :page, :per_page])
  @query = search_params[:q] || '*'
  @options = options.symbolize_keys
  @block = block
end

def documents
  @documents ||= Sunspot.search(searchable) do
    fulltext query
    paginate search_pagination
    order_by *search_sort

    # Call the block if given, which allows for further filtering/overriding
    instance_exec(&block) if block.present?
  end
end
