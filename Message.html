def email_to!(user, **params)
  # Ensure keys are symbols
  params.symbolize_keys!

  # Start with the default subject/body text, but allow each to be overridden with provided params
  # Also allows for additional params that will be accessible in the mail template as @ vars
  message = { subject: subject, body: body }.merge(params)

  # Remove the subject/body from the params so when interpolating, the user cannot
  # accidentally include "self" inside of "self"
  params.reject! { |k, _| %i[subject body].include?(k) }

  # Exit if the subject or body is blank at this point
  raise InvalidParameter, 'Email subject and body cannot be blank'._ if message[:subject].blank? || message[:body].blank?

  # Format the message body (adds <p> tags to paragraphs)
  message[:body] = simple_format(message[:body] % params)

  # Apply interpolation to the subject, so variable text can be introduced via the params hash
  message[:subject] = message[:subject] % params

  # Send the notification
  user.notifications.create { |n| n.email message[:subject], message }
end
