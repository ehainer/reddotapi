def facilities(inactive)
  results = inactive ? occupancies.active : occupancies
  results = results.joins(unit: [:facility]).group('facilities.id, occupancies.id').order('occupancies.id ASC').pluck('facilities.id').uniq
  Facility.find(results)
end

def reserve!(unit, move_in, **args)
  args.slice!(:moved_out_at, :due_day)
  occupancies.create!(args.merge({ unit: unit, moved_in_at: move_in, reserved_at: Time.current }))
end

def units(inactive, *facilities)
  # Form array of facility ids to filter by
  facilities = facilities.flatten.map { |facility| facility.is_a?(Facility) ? facility.id : facility } 

  results = inactive ? occupancies.active.includes(unit: [:facility]) : occupancies.includes(unit: [:facility])
  results = results.where('facilities.id' => facilities) unless facilities.empty?
  results = results.pluck('units.id').uniq
  Unit.find(results)
end
