def create
  authorize! :manage, Reminder
  @reminder = current_user.reminders.build(reminder_params)

  if @reminder.save
    render partial: 'reminders/show/reminder', locals: { reminder: @reminder.decorate }
  else
    flash! alert: @reminder.errors.full_messages
    head :bad_request
  end
end
